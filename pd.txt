#include <stdio.h>
#include <string.h>
int teste = 0;
// Função para calcular o comprimento da subsequência comum mais longa (LCS)
int lcs_dp(char *X, char *Y) {
    int m = strlen(X);
    int n = strlen(Y);
    int L[m+1][n+1];  // Matriz para armazenar os comprimentos das subsequências

    // Preenche a matriz L usando programação dinâmica
    for (int i = 0; i <= m; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == 0 || j == 0) {
                L[i][j] = 0;  // Base de recursão: LCS de uma string vazia é 0
            } else if (X[i-1] == Y[j-1]) {
                L[i][j] = L[i-1][j-1] + 1;  // Se os caracteres são iguais, aumenta o comprimento
            } else {
                L[i][j] = (L[i-1][j] > L[i][j-1]) ? L[i-1][j] : L[i][j-1];  // Caso contrário, pega o máximo
            }
        }
    }

    // O comprimento da LCS estará na célula L[m][n]
    return L[m][n];
}

// Função para ler o arquivo de entrada e passar para a função LCS
void process_file(char *filename) {
    FILE *file = fopen(filename, "r");
    if (!file) {
        printf("Erro ao abrir o arquivo.\n");
        return;
    }

    char X[teste+1], Y[teste+1];  // Supondo que o tamanho das strings seja 100
    // Lê as duas linhas de entrada
    if (fgets(X, sizeof(X), file) && fgets(Y, sizeof(Y), file)) {
        // Remover o caractere de nova linha que pode ser lido pelo fgets
        X[strcspn(X, "\n")] = 0;
        Y[strcspn(Y, "\n")] = 0;

        // Chama a função LCS para calcular o comprimento da LCS
        printf("Comprimento da LCS: %d\n", lcs_dp(X, Y));
    } else {
        printf("Erro ao ler as strings do arquivo.\n");
    }

    fclose(file);
}

int main() {
    char filename[] = "entrada.txt";  // Substitua pelo nome do arquivo de entrada
    teste = 25;
    process_file(filename);
    return 0;
}
